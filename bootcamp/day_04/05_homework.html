n = 1000
data = dict(
    age = np.clip(rng.normal(50,20,size=(n,)).astype(int),0,100),
    sex = rng.choice(['M','F'],size=(n,)),
    height = np.clip(np.round(rng.normal(1.5,0.2,size=(n,)),2),1.2,2.1),
